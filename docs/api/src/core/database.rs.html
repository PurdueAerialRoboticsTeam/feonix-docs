18pub async fn create_schema(pool: &PgPool) -> anyhow::Result<()> {
19    sqlx::query(
20        r#"
21CREATE TABLE IF NOT EXISTS gps (
22    timestamp INTEGER PRIMARY KEY CHECK (timestamp >= 0),
23    lat REAL NOT NULL,
24    long REAL NOT NULL,
25    alt REAL NOT NULL,
26    heading REAL NOT NULL
27);
28"#,
29    )
30    .execute(pool)
31    .await?;
32
33    sqlx::query(
34        r#"
35CREATE TABLE IF NOT EXISTS images (
36    filepath TEXT PRIMARY KEY,
37    timestamp INTEGER NOT NULL CHECK (timestamp >= 0)
38);
39"#,
40    )
41    .execute(pool)
42    .await?;
43
44    sqlx::query(
45        r#"
46CREATE TABLE IF NOT EXISTS geotags (
47    filepath TEXT PRIMARY KEY REFERENCES images (filepath) ON UPDATE CASCADE ON DELETE CASCADE,
48    image_timestamp INTEGER NOT NULL CHECK (image_timestamp >= 0),
49    gps_time INTEGER NOT NULL REFERENCES gps (timestamp) ON UPDATE CASCADE ON DELETE RESTRICT,
50    lat REAL NOT NULL,
51    long REAL NOT NULL,
52    alt REAL NOT NULL,
53    heading REAL NOT NULL
54);
55"#,
56    )
57    .execute(pool)
58    .await?;
59
60    sqlx::query(
61        r#"
62CREATE TABLE IF NOT EXISTS objects (
63    id INTEGER PRIMARY KEY,
64    original_filepath TEXT NOT NULL,
65    lat REAL NOT NULL,
66    long REAL NOT NULL,
67    class INTEGER NOT NULL,
68    max_confidence REAL NOT NULL,
69    num_detections INTEGER NOT NULL,
70    UNIQUE(lat, long, class)
71);
72"#,
73    )
74    .execute(pool)
75    .await?;
76    Ok(())
77}

79pub async fn get_gps(pool: &PgPool, timestamp: &i64) -> anyhow::Result<(GPSRow, GPSRow)> {
80    // Fetch closest 2 timestamps by their distance to the camera timestamp
81    let row = sqlx::query_as::<_, GPSRow>("SELECT * FROM gps ORDER BY ABS(timestamp - ?) LIMIT 2")
82        .bind(timestamp)
83        .fetch_all(pool)
84        .await?;
85
86    Ok((row[0].clone(), row[1].clone()))
87}

103async fn bulk_upsert_objects(
104    // inserts all the objects, or updated them if they already exist
105    pool: &sqlx::PgPool,
106    objects: &[ObjectRow],
107) -> anyhow::Result<()> {
108    // deconstruct the objects to insert
109    let mut lats = Vec::with_capacity(objects.len());
110    let mut longs = Vec::with_capacity(objects.len());
111    let mut classes = Vec::with_capacity(objects.len());
112    let mut max_confidences = Vec::with_capacity(objects.len());
113    let mut num_detections = Vec::with_capacity(objects.len());
114    let mut original_filepaths = Vec::with_capacity(objects.len());
115
116    for obj in objects {
117        lats.push(obj.lat);
118        longs.push(obj.long);
119        classes.push(obj.class);
120        max_confidences.push(obj.max_confidence);
121        num_detections.push(obj.num_detections);
122        original_filepaths.push(&obj.original_filepath);
123    }
124
125    // bulk upsert query. (either replace if it exists, or insert if not)
126    sqlx::query(
127        r#"
128            INSERT INTO objects (lat, long, class, max_confidence, num_detections, original_filepath)
129            SELECT * FROM UNNEST(
130                $1::real[],
131                $2::real[],
132                $3::integer[],
133                $4::real[],
134                $5::integer[],
135                &6::text[]
136            )
137            ON CONFLICT (lat, long, class) DO UPDATE
138            SET
139                max_confidence = excluded.max_confidence,
140                num_detections = objects.num_detections + 1;
141        "#,
142    )
143    .bind(lats)
144    .bind(longs)
145    .bind(classes)
146    .bind(max_confidences)
147    .bind(num_detections)
148    .bind(original_filepaths)
149    .execute(pool)
150    .await?;
151
152    Ok(())
153}

155pub async fn smart_update_database(
156    pool: &PgPool,
157    objects: Vec<Object>,
158    file_path: &String,
159) -> anyhow::Result<()> {
160    // detects similarly placed objects (withing 11m currently) and then tracks them as an object in the database increasing label confidence and count
161    let current_objects = sqlx::query_as::<_, ObjectRow>("SELECT * from objects")
162        .fetch_all(pool)
163        .await?;
164
165    let mut new_objects = Vec::<ObjectRow>::new();
166
167    let object_tolerance = get_env("OBJECT_CORD_TOLERANCE", 0.0001_f64).abs();
168
169    for object in objects.iter() {
170        let mut found = false;
171        for current_object in current_objects.iter() {
172            if (current_object.lat - object.lat).abs() < object_tolerance
173                && (current_object.long - object.long).abs() < object_tolerance
174                && object.class == current_object.class
175            {
176                // TODO Create other object detection clustering algoirthms (might have to rework the current db to also store all detection if a clustering alg is used or similar).
177                new_objects.push(ObjectRow {
178                    num_detections: current_object.num_detections + 1,
179                    max_confidence: f32::max(current_object.max_confidence, object.confidence),
180                    ..(*current_object).clone()
181                });
182                found = true;
183            }
184        }
185        if !found {
186            new_objects.push(ObjectRow {
187                lat: object.lat,
188                long: object.long,
189                class: object.class,
190                max_confidence: object.confidence,
191                num_detections: 1,
192                original_filepath: file_path.clone(),
193            });
194        }
195    }
196
197    bulk_upsert_objects(pool, &new_objects).await?;
198
199    Ok(())
200}

202pub async fn start_postgres_container() -> anyhow::Result<ContainerAsync<GenericImage>> {
203    Ok(GenericImage::new("postgres", "latest")
204        .with_wait_for(WaitFor::Duration {

213        .with_env_var("POSTGRES_USER", "user")
214        .with_env_var("POSTGRES_PASSWORD", "password")
215        .start()
216        .await?)

219pub async fn start_postgres_embedded() -> anyhow::Result<(PostgreSQL, String)> {

223    let mut embedded_postgresql = PostgreSQL::new(settings);
224
225    embedded_postgresql.setup().await?;
226    embedded_postgresql.start().await?;
227    let database_name = "local";
228    embedded_postgresql.create_database(database_name).await?;
229    let url = embedded_postgresql.settings().url(database_name);
